Как можем позиционировать элементы?

position: static


.box {
    position: static;
}


.box {
    position: static;
    top: 70px;
    left: 70px;
}

position: relative


.box {
    position: relative;
}


.box {
    position: relative;
    top: 70px;
    left: 70px;
}

position: sticky


.box {
    position: relative;
    top: 70px;
    left: 70px;
}


.box {
    position: sticky;
    top: 70px;
    left: 70px;
}

position: absolute


.box {
    position: sticky;
    top: 70px;
    left: 70px;
}


.box {
    position: absolute;
    top: 70px;
    left: 70px;
}

position: fixed


.box {
    position: absolute;
    top: 70px;
    left: 70px;
}


.box {
    position: fixed;
    top: 70px;
    left: 70px;
}
