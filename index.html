🤖 JavaScript

            
            import { 
                computePosition, offset, autoPlacement, autoUpdate 
            } from '@floating-ui/dom';

            const updatePosition = () => {
                computePosition(a, b, {
                    middleware: [
                        offset({ mainAxis: -100, crossAxis: 100 }), 
                        autoPlacement(),
                    ],
                }).then(({x, y}) => {
                    Object.assign(b.style, {
                        top: `${y}px`,
                        left: `${x}px`,
                    });
                });
            };

            autoUpdate(a, b, updatePosition);
